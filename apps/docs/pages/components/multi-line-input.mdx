# MultiLine Input

Textarea variant of `<Input>`. Source: Figma `FKInput` Feature=Multi Line
(`12:68514`). Optional auto-resize + character counter.

## Live preview

<Demo />

export const Demo = () => {
  const [notes, setNotes] = useState('');
  const [msg, setMsg] = useState('');
  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: 20, padding: 24, background: 'var(--semantic-level-basement, #0b0f1a)', borderRadius: 12, marginBottom: 24, maxWidth: 480 }}>
      <MultiLineInput
        label="Order notes"
        placeholder="Anything worth noting…"
        value={notes}
        onChange={(e) => setNotes(e.target.value)}
        rows={3}
        fullWidth
      />
      <MultiLineInput
        label="Feedback (auto-grows)"
        placeholder="Type more to see it grow"
        value={msg}
        onChange={(e) => setMsg(e.target.value)}
        rows={2}
        maxRows={8}
        autoResize
        showCounter
        maxLength={280}
        fullWidth
      />
    </div>
  );
};
